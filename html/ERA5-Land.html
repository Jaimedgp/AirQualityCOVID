## derivada de la temperatura a 2 m. y la temperatura del punto de rocío, de acuerdo con la fórmula de Magnus: https://earthscience.stackexchange.com/questions/16570/how-to-calculate-relative-humidity-from-temperature-dew-point-and-pressure
load("./rh_daily_2010_2020.rda")
rh.sea = lapply(list(c(12, 1, 2), 3:5, 6:8, 9:11), function(sea) {
    climatology(subsetGrid(rh, season = sea))
})
spatialPlot(makeMultiGrid(rh.sea, skip.temporal.check = T), backdrop.theme = "coastline",
           layout = c(2,2), as.table = T, set.min = 0, set.max = 100, at = seq(0, 100, 100/20), 
         names.attr = c("DJF", "MAM", "JJA", "SON"), main = "Relative Humidity (%)")
